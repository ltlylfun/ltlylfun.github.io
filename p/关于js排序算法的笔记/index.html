工具函数

首先，我们定义一个用于交换数组元素的工具函数：

1function swap(arr, i, j) {
2  [arr[i], arr[j]] = [arr[j], arr[i]];
3}

快速排序 (Quick Sort)

  • 时间复杂度: 平均 O(nlogn)，最坏 O(n²)
  • 空间复杂度: O(logn)
  • 稳定性: 不稳定
 1function quickSort(arr, left = 0, right = arr.length - 1) {
 2  if (left < right) {
 3    const pivot = partition(arr, left, right);
 4    quickSort(arr, left, pivot - 1);
 5    quickSort(arr, pivot + 1, right);
 6  }
 7  return arr;
 8}
 9
10function partition(arr, left, right) {
11  const pivot = arr[right];
12  let i = left - 1;
13
14  for (let j = left; j < right; j++) {
15    if (arr[j] < pivot) {
16      i++;
17      swap(arr, i, j);
18    }
19  }
20  swap(arr, i + 1, right);
21  return i + 1;
22}

冒泡排序 (Bubble Sort)

  • 时间复杂度: O(n²)
  • 空间复杂度: O(1)
  • 稳定性: 稳定
 1function bubbleSort(arr) {
 2  const n = arr.length;
 3  for (let i = 0; i < n - 1; i++) {
 4    let swapped = false;
 5    for (let j = 0; j < n - 1 - i; j++) {
 6      if (arr[j] > arr[j + 1]) {
 7        swap(arr, j, j + 1);
 8        swapped = true;
 9      }
10    }
11    if (!swapped) break; // 如果没有交换，说明已经排序完成
12  }
13  return arr;
14}

插入排序 (Insertion Sort)

  • 时间复杂度: O(n²)
  • 空间复杂度: O(1)
  • 稳定性: 稳定
 1function insertionSort(arr) {
 2  for (let i = 1; i < arr.length; i++) {
 3    const key = arr[i];
 4    let j = i - 1;
 5    while (j >= 0 && arr[j] > key) {
 6      arr[j + 1] = arr[j];
 7      j--;
 8    }
 9    arr[j + 1] = key;
10  }
11  return arr;
12}

归并排序 (Merge Sort)

  • 时间复杂度: O(nlogn)
  • 空间复杂度: O(n)
  • 稳定性: 稳定
 1function mergeSort(arr) {
 2  if (arr.length <= 1) return arr;
 3
 4  const mid = Math.floor(arr.length / 2);
 5  const left = mergeSort(arr.slice(0, mid));
 6  const right = mergeSort(arr.slice(mid));
 7
 8  return merge(left, right);
 9}
10
11function merge(left, right) {
12  const result = [];
13  let i = 0,
14    j = 0;
15
16  while (i < left.length && j < right.length) {
17    if (left[i] <= right[j]) {
18      result.push(left[i++]);
19    } else {
20      result.push(right[j++]);
21    }
22  }
23
24  return result.concat(left.slice(i)).concat(right.slice(j));
25}

堆排序 (Heap Sort)

  • 时间复杂度: O(nlogn)
  • 空间复杂度: O(1)
  • 稳定性: 不稳定
 1function heapSort(arr) {
 2  const n = arr.length;
 3
 4  // 构建最大堆
 5  for (let i = Math.floor(n / 2) - 1; i >= 0; i--) {
 6    heapify(arr, n, i);
 7  }
 8
 9  // 一个个从堆顶取出最大值
10  for (let i = n - 1; i > 0; i--) {
11    swap(arr, 0, i);
12    heapify(arr, i, 0);
13  }
14
15  return arr;
16}
17
18function heapify(arr, n, i) {
19  let largest = i;
20  const left = 2 * i + 1;
21  const right = 2 * i + 2;
22
23  if (left < n && arr[left] > arr[largest]) {
24    largest = left;
25  }
26
27  if (right < n && arr[right] > arr[largest]) {
28    largest = right;
29  }
30
31  if (largest !== i) {
32    swap(arr, i, largest);
33    heapify(arr, n, largest);
34  }
35}

桶排序 (Bucket Sort)

  • 时间复杂度: 平均 O(n + k)，最坏 O(n²)
  • 空间复杂度: O(n + k)
  • 稳定性: 稳定
 1function bucketSort(arr, bucketSize = 5) {
 2  if (arr.length === 0) return arr;
 3
 4  // 找出最大值和最小值
 5  let minValue = arr[0];
 6  let maxValue = arr[0];
 7  for (let i = 1; i < arr.length; i++) {
 8    if (arr[i] < minValue) {
 9      minValue = arr[i];
10    } else if (arr[i] > maxValue) {
11      maxValue = arr[i];
12    }
13  }
14
15  // 创建桶
16  const bucketCount = Math.floor((maxValue - minValue) / bucketSize) + 1;
17  const buckets = new Array(bucketCount);
18  for (let i = 0; i < buckets.length; i++) {
19    buckets[i] = [];
20  }
21
22  // 将数值分配到各个桶中
23  for (let i = 0; i < arr.length; i++) {
24    const bucketIndex = Math.floor((arr[i] - minValue) / bucketSize);
25    buckets[bucketIndex].push(arr[i]);
26  }
27
28  // 对每个桶内部进行排序，这里使用插入排序
29  const result = [];
30  for (let i = 0; i < buckets.length; i++) {
31    insertionSort(buckets[i]);
32    result.push(...buckets[i]);
33  }
34
35  return result;
36}

基数排序 (Radix Sort)

  • 时间复杂度: O(n * k)，k 是最大数字的位数
  • 空间复杂度: O(n + k)
  • 稳定性: 稳定
 1function radixSort(arr) {
 2  if (arr.length === 0) return arr;
 3
 4  // 找出最大值
 5  let maxValue = arr[0];
 6  for (let i = 1; i < arr.length; i++) {
 7    if (arr[i] > maxValue) {
 8      maxValue = arr[i];
 9    }
10  }
11
12  // 对每一位进行计数排序
13  let exp = 1;
14  while (maxValue / exp > 0) {
15    countingSortByDigit(arr, exp);
16    exp *= 10;
17  }
18
19  return arr;
20}
21
22function countingSortByDigit(arr, exp) {
23  const output = new Array(arr.length).fill(0);
24  const count = new Array(10).fill(0);
25
26  // 统计每个数字出现的次数
27  for (let i = 0; i < arr.length; i++) {
28    count[Math.floor(arr[i] / exp) % 10]++;
29  }
30
31  // 计算累积数量
32  for (let i = 1; i < 10; i++) {
33    count[i] += count[i - 1];
34  }
35
36  // 构建输出数组
37  for (let i = arr.length - 1; i >= 0; i--) {
38    const digit = Math.floor(arr[i] / exp) % 10;
39    output[count[digit] - 1] = arr[i];
40    count[digit]--;
41  }
42
43  // 将输出数组复制回原数组
44  for (let i = 0; i < arr.length; i++) {
45    arr[i] = output[i];
46  }
47}

计数排序 (Counting Sort)

  • 时间复杂度: O(n + k)，k 是数值范围
  • 空间复杂度: O(k)
  • 稳定性: 稳定
 1function countingSort(arr) {
 2  if (arr.length === 0) return arr;
 3
 4  // 找出最大值和最小值
 5  let maxValue = arr[0];
 6  let minValue = arr[0];
 7  for (let i = 1; i < arr.length; i++) {
 8    if (arr[i] > maxValue) maxValue = arr[i];
 9    if (arr[i] < minValue) minValue = arr[i];
10  }
11
12  const range = maxValue - minValue + 1;
13  const count = new Array(range).fill(0);
14  const output = new Array(arr.length);
15
16  // 统计每个元素出现的次数
17  for (let i = 0; i < arr.length; i++) {
18    count[arr[i] - minValue]++;
19  }
20
21  // 计算累积数量
22  for (let i = 1; i < range; i++) {
23    count[i] += count[i - 1];
24  }
25
26  // 构建输出数组
27  for (let i = arr.length - 1; i >= 0; i--) {
28    output[count[arr[i] - minValue] - 1] = arr[i];
29    count[arr[i] - minValue]--;
30  }
31
32  return output;
33}
